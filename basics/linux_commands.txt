ls = показывает содержание папки (файлы и другие папки)
ls название_директории{py33} = показывает все файлы и папки внутри папки py33
ls -a = показыавет скрытые файлы и папки
ls -l = показывает дополнительную информацию о файлах и директориях
ls -la = показывает скрытые файлы и директории с подробной информацией 

cd = переход домашнюю директорию 
cd ~ = переход домашнюю директорию
сd test = переход в папку test, если такой директории нет, то выйдет ошибка>
cd .. = переход на директорию выше
cd ../.. = переход на 2 директории выше
cd - = cntr + z

mkdir test = создает папку test в текущей тиректорит
mkdir test/text2 = создает папку в test2 внутри папки tes t, если папка test существует 
mkdie test test2 = создает папку test и test2 в текущей директории

touch file.txt = создает файл file с разрешением txt
touch file.txt file2.txt = создает файл file1.txt и file2.txt

rm test.txt = удаляет файл test.txt
rmdir test = удаляет пустую папку test
rm -rf = удаляет всё: файлы, директории с содержимым, пустые директории. Рекурсивное удаление(полное)  

mv test.txt test = перемещает файл text.txt в директорию test
mv test1.txt test2.txt = переименовает файл test1.txt на test2.txt

pwd(print working directory) = показывет абсолютный путь (начиная скорневой папки /) до текущей директории
history = показывет все команды, которые мы вводили в данный терминал
history 10 =  показывает 10 послдених введенных команд из истоии команд
history | grep test = показывает команды из историии команд содержавщие слово test

nano test.txt = открывает файл test.txt в текстовом редакторе nano
cat test.txt = выводит содержимое файла test.txt в терминал

sudo(super user do) = права суперюзера
apt = системный установшик пакетов
brew = установщик пакетов (macos)
